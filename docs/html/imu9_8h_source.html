211 static inline void caerIMU9EventValidate(caerIMU9Event event, caerIMU9EventPacket packet) {
212  if (!caerIMU9EventIsValid(event)) {
213  SET_NUMBITS32(event->info, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
214 
215  // Also increase number of events and valid events.
216  // Only call this on (still) invalid events!
217  caerEventPacketHeaderSetEventNumber(&packet->packetHeader,
218  caerEventPacketHeaderGetEventNumber(&packet->packetHeader) + 1);
219  caerEventPacketHeaderSetEventValid(&packet->packetHeader,
220  caerEventPacketHeaderGetEventValid(&packet->packetHeader) + 1);
221  }
222  else {
223  caerLog(CAER_LOG_CRITICAL, "IMU9 Event", "Called caerIMU9EventValidate() on already valid event.");
224  }

236 static inline void caerIMU9EventInvalidate(caerIMU9Event event, caerIMU9EventPacket packet) {
237  if (caerIMU9EventIsValid(event)) {
238  CLEAR_NUMBITS32(event->info, VALID_MARK_SHIFT, VALID_MARK_MASK);
