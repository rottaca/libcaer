359 
360  CAER_EVENT_PACKET_CONTAINER_CONST_ITERATOR_START(container)

362  if (caerEventPacketHeaderGetEventNumber(caerEventPacketContainerIteratorElement) == 0) {
363  continue;
364  }
365

367  const void *firstEvent = caerGenericEventGetEvent(caerEventPacketContainerIteratorElement, 0);
368  int64_t currLowestEventTimestamp = caerGenericEventGetTimestamp64(firstEvent, caerEventPacketContainerIteratorElement);
369 
370  const void *lastEvent = caerGenericEventGetEvent(caerEventPacketContainerIteratorElement,
371  caerEventPacketHeaderGetEventNumber(caerEventPacketContainerIteratorElement) - 1);
372  int64_t currHighestEventTimestamp = caerGenericEventGetTimestamp64(lastEvent, caerEventPacketContainerIteratorElement);
373 
374  // Update tracked timestamps (or initialize if needed).
375  if ((lowestTimestamp == -1) || (lowestTimestamp > currLowestEventTimestamp)) {
376  lowestTimestamp = currLowestEventTimestamp;

380  highestTimestamp = currHighestEventTimestamp;
381  }

384  eventsValid += caerEventPacketHeaderGetEventValid(caerEventPacketContainerIteratorElement);
