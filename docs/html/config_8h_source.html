211 static inline void caerConfigurationEventValidate(caerConfigurationEvent event, caerConfigurationEventPacket packet) {
212  if (!caerConfigurationEventIsValid(event)) {
213  SET_NUMBITS8(event->moduleAddress, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
214

224  "Called caerConfigurationEventValidate() on already valid event.");
