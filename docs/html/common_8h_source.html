525 static inline bool caerEventPacketEquals(caerEventPacketHeaderConst firstPacket,
526  caerEventPacketHeaderConst secondPacket) {
527  // If any of the packets is NULL, they can't be equal.
528  if (firstPacket == NULL || secondPacket == NULL) {
529  return (false);
530  }
531 
532  // If both packets are the same packet (pointer equal),
533  // they are of course indeed equal packets.
534  if (firstPacket == secondPacket) {
535  return (true);
536  }
537 
538  // Actually compare memory now. We compare header equality, and
539  // all events up to eventNumber. The remaining events up to
540  // eventCapacity are by definition all zeroed out, so must be
541  // equal, if the capacity is the same, which it is, as we check
542  // for that when ensuring header equality.
543  if (memcmp(firstPacket, secondPacket, CAER_EVENT_PACKET_HEADER_SIZE) != 0) {
544  return (false);
545  }
546 
547  size_t memCmpSize = (size_t) (caerEventPacketHeaderGetEventNumber(firstPacket)
548  * caerEventPacketHeaderGetEventSize(firstPacket));
549  if (memcmp(((const uint8_t *) firstPacket) + CAER_EVENT_PACKET_HEADER_SIZE,
550  ((const uint8_t *) secondPacket) + CAER_EVENT_PACKET_HEADER_SIZE, memCmpSize) != 0) {
551  return (false);
552  }
553 
554  return (true);
555 }

587 static inline void caerEventPacketClean(caerEventPacketHeader packet) {
588  // Handle empty event packets.
589  if (packet == NULL) {
590  return;
591  }
592 
593  // Calculate needed memory for new event packet.
594  int32_t eventValid = caerEventPacketHeaderGetEventValid(packet);
595  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(packet);
596 
597  // If we have no invalid events, we're already done.
598  if (eventValid == eventNumber) {
599  return;
600  }
601 
602  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
603  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
604 
605  // Move all valid events close together. Must check every event for validity!
606  size_t offset = CAER_EVENT_PACKET_HEADER_SIZE;
607 
608  CAER_ITERATOR_VALID_START(packet, const void *)
609  void *dest = ((uint8_t *) packet) + offset;
610 
611  if (dest != caerIteratorElement) {
612  memcpy(dest, caerIteratorElement, (size_t) eventSize);
613  offset += (size_t) eventSize;
614  }
615  }
616 
617  // Reset remaining memory, up to capacity, to zero (all events invalid).
618  memset(((uint8_t *) packet) + offset, 0, (size_t) ((eventCapacity - eventValid) * eventSize));
619 
620  // Event capacity remains unchanged, event number shrunk to event valid number.
621  caerEventPacketHeaderSetEventNumber(packet, eventValid);
622 }

642 static inline caerEventPacketHeader caerEventPacketResize(caerEventPacketHeader packet, int32_t newEventCapacity) {
643  if (packet == NULL || newEventCapacity <= 0) {
644  return (NULL);
645  }
646 
647  // Always clean for consistency with shrink case (side-effects guarantee).
648  caerEventPacketClean(packet);
649 
650  int32_t oldEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
651 
652  if (oldEventCapacity == newEventCapacity) {
653  // Nothing to do in this case.
654  return (packet);
655  }
656 
657  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
658  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (newEventCapacity * eventSize);
659 
660  // Reallocate memory used to hold events.
661  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
662  if (packet == NULL) {
663  caerLog(CAER_LOG_CRITICAL, "Event Packet",
664  "Failed to reallocate %zu bytes of memory for resizing Event Packet of capacity %"
665  PRIi32 " to new capacity of %" PRIi32 ". Error: %d.", newEventPacketSize, oldEventCapacity,
666  newEventCapacity, errno);
667  return (NULL);
668  }
669 
670  if (newEventCapacity > oldEventCapacity) {
671  // Capacity increased: we simply zero out the newly added events.
672  size_t oldEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (oldEventCapacity * eventSize);
673 
674  memset(((uint8_t *) packet) + oldEventPacketSize, 0,
675  (size_t) ((newEventCapacity - oldEventCapacity) * eventSize));
676  }
677  else {
678  // Capacity decreased: the events were cleaned, so eventValid == eventNumber.
679  // They also are all together in memory. Thus we can simply keep the current
680  // eventValid/eventNumber counts if the capacity is still bigger or equal to
681  // them, or, if new capacity is smaller, we reset them to that value.
682  int32_t oldEventNumber = caerEventPacketHeaderGetEventNumber(packet);
683 
684  if (newEventCapacity < oldEventNumber) {
685  caerEventPacketHeaderSetEventValid(packet, newEventCapacity);
686  caerEventPacketHeaderSetEventNumber(packet, newEventCapacity);
687  }
688  }
689 
690  // Update capacity header field.
691  caerEventPacketHeaderSetEventCapacity(packet, newEventCapacity);
692 
693  return (packet);
694 }
